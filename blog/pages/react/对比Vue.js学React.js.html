# 对比 Vue.js 学 React.js

# 使用脚手架创建项目

# @vue/cli

# 安装 vue-cli 脚手架
npm install -g @vue/cli
# OR
yarn global add @vue/cli

# 创建 my-app 项目
vue create my-app

# create-react-app

# 安装 create-react-app 脚手架并创建 my-app 项目
npx create-react-app my-app

#OR
npm install -g create-react-app
# 创建 my-app 项目
yarn create react-app my-app
# OR
npm init react-app my-app

# 组件

# Vue 组件

  1. template 编写组件 HTML 模板， 只有一个根节点
  2. script 编写 js， 必须默认导出
  3. style 编写 css
  4. 初始化数据状态，统一放data的返回值里
  5. 更新状态： this.size = 'middle'

vue 单文件组件示例

<template>
  <div id="app">
    <img alt="Vue logo" src="./assets/logo.png" />
    <!-- 调用 HelloWorld 组件 传递 msg 信息-->
    <HelloWorld msg="Welcome to Your Vue.js App" />
  </div>
</template>

<script>
// 导入 HelloWorld 组件
import HelloWorld from './components/HelloWorld.vue';
// 默认导出
export default {
  // 声明当前组件名称
  name: 'App',
  // 声明引入的组件
  components: {
    HelloWorld,
  },
  data() {
    return {
      size: 'small'
    }
  }
  methods: {
    changeSize() {
      this.size = 'middle';
    }
  }
};
</script>

# React 组件

# 书写 jsx 注意事项

  1. jsx 中元素的class要写成 className， 避免和 js 中的 class 类 混淆
  2. label 元素的for属性要写成htmlFor， 避免和 js 中的 for 循环混淆
  3. 元素绑定事件, vue中为@click, @change等; React中为> onClick, onChange
  4. 调用变量需要使用{var}, 如 <img src={logo}/>

# 函数式组件

  1. 函数式组件必须返回jsx
  2. 使用 useState 设置和更新状态
  3. 使用方式: const [size, setSize] = useState('small');, useState() 会返回一个长度为 2 的数组 array ， size (即array[0]) 为当前状态， setSize(即array[1])为更新状态的方法;
  4. 更新状态: const changeSize = () => setSize('middle');

函数式组件示例

import logo from './logo.svg';
import './App.css';
import { useState } from 'react';
function App() {
  const [size, setSize] = useState('small');
  const changeSize = () => setSize('middle');
  return (
    <div className="App">
      <header className="App-header">
        <img src={logo} className="App-logo" alt="logo" />
        <button onClick={changeSize}>...</button>
      </header>
    </div>
  );
}
export default App;

# class 类组件

  1. 组件必须继承自React.component(可以使用import { component } from 'react'; 达到简写的目的)
  2. 组件必须有 render 函数并返回 jsx
  3. 组件中存在constructor 构造器时， 必须有super(props)
  4. 在constructor中初始化状态： this.state = {size: 'small'}
  5. 更新状态： this.setState = {size: 'middle'};
  6. 绑定事件需要绑定this, 有如下几种方式：
// 1. 使用`bind`绑定
<button onClick={this.changeSize.bind(this)}>调节大小</button>
// 2. 使用箭头函数， 无需绑定
<button onClick={() => this.changeSize()}>调节大小</button>
// 3. 在 constructor 中绑定this
constructor(props) {
  super(props);
  this.state = {
    size: 'small'
  };
  this.changeSize = this.changeSize.bind(this);
}

类组件示例

import logo from './logo.svg';
import './App.css';
import React from 'react';

class App extends React.component {
  constructor(props) {
    super(props);
    this.state = {
      size: 'small'
    };
  }
  changeSize() {
    this.setState({
      size: 'middle'
    })
  }
  render (
    <div className="App">
      <button onClick={this.changeSize.bind(this)}>...</button>
   </div>
  )
}

// 默认导出
export default App;
